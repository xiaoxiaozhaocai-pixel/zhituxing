### 任务 1：修复限流 bug（src/middleware.ts）🔴 P0

**现状**：全局 IP 限流 100 次/分钟，触发后冷却 60s，导致：
1. 静态资源（`_next/static/*`、favicon、图片、字体）也参与限流，正常用户秒被打爆
2. Zeabur 反向代理拿到的 IP 是边缘节点，多用户共享同一桶
3. 阈值 100 太低，登录页一刷 5-10 个 API 就触发

**修改要求**：

```ts
// src/middleware.ts

// 1. 阈值从 100 改 400/min
const RATE_LIMIT = 400;
const RATE_WINDOW_MS = 60_000;

// 2. 静态资源豁免（在 middleware 函数最顶部加）
const pathname = request.nextUrl.pathname;
const STATIC_PREFIXES = ['/_next/static', '/_next/image', '/favicon', '/images/', '/fonts/', '/assets/'];
const STATIC_EXTENSIONS = ['.ico', '.png', '.jpg', '.jpeg', '.gif', '.svg', '.webp', '.woff', '.woff2', '.ttf', '.css', '.js', '.map'];
if (
  STATIC_PREFIXES.some(p => pathname.startsWith(p)) ||
  STATIC_EXTENSIONS.some(ext => pathname.endsWith(ext))
) {
  return NextResponse.next();
}

// 3. 优先按 user_id 限流（cookie 里有 sb-access-token 就解 JWT 取 sub）
// 如果是匿名用户回退按 IP 限流
function getRateLimitKey(req: NextRequest): string {
  const token = req.cookies.get('sb-access-token')?.value;
  if (token) {
    try {
      const payload = JSON.parse(atob(token.split('.')[1]));
      if (payload?.sub) return `user:${payload.sub}`;
    } catch {}
  }
  return `ip:${getClientIP(req)}`;
}

// 4. getClientIP 修正：Zeabur 在 x-forwarded-for 里塞多个 IP，取第一个真实 IP
function getClientIP(req: NextRequest): string {
  const xff = req.headers.get('x-forwarded-for');
  if (xff) {
    // 取第一个非空非 unknown 的 IP
    const ips = xff.split(',').map(s => s.trim()).filter(s => s && s !== 'unknown');
    if (ips[0]) return ips[0];
  }
  return req.headers.get('x-real-ip') ||
         req.headers.get('cf-connecting-ip') ||
         'anonymous';
}
```

**验收**：
- 本地启动 `pnpm dev`，浏览器疯狂刷新首页 20 次，不能出现 429
- 登录后调用 5+ 个 API 不能 429

```ts
// 1. 加 in-memory 缓存（5 分钟 TTL）
const quotaCache = new Map<string, { data: any; expires: number }>();
const CACHE_TTL_MS = 5 * 60 * 1000;

export async function GET(request: NextRequest) {
  const user = await getCurrentUser(request);
  if (!user) return NextResponse.json({ error: '请先登录' }, { status: 401 });

  const cached = quotaCache.get(user.id);
  if (cached && cached.expires > Date.now()) {
    return NextResponse.json(cached.data);
  }

  // ... 原有的查询 user_profiles + user_quotas 逻辑 ...

  // 2. 返回字段补齐
  const responseData = {
    success: true,
    data: {
      userType: profile.user_type,
      quota: q.quota,
      usedQuota: q.used_quota,
      monthlyQuota: q.monthly_quota,    // ← 补
      monthlyUsed: q.used_quota,        // ← 补（暂复用 used_quota）
      interviewQuota: q.interview_quota,
      assessmentQuota: q.assessment_quota,
      memberExpiresAt: q.member_expires_at,
      quotaResetTime: q.quota_reset_time,
    },
  };

  // 3. 写缓存
  quotaCache.set(user.id, { data: responseData, expires: Date.now() + CACHE_TTL_MS });
  return NextResponse.json(responseData);
}

// 4. POST/PATCH 等修改操作后必须清缓存
function invalidateQuotaCache(userId: string) {
  quotaCache.delete(userId);
}
```
